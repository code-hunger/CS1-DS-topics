# Обхождане на графи

## В ширина

- r ∈ V - начален връх
- adj = списък съседи
- eyes = {r} // тип опашка - слагаме накрая, взимаме първия
- dist = V → N
- dist[r] = 0

- while #eyes > 0:
    - w ← eyes
    - eyes ← eyes - { w }

    - visit w

    - ∀w' ∈ adj[w]:
        - if dist[w'] = ∞:
            - dist[w'] ← dist[w] + 1
            - eyes ← eyes ∪ { w' }

## В дълбочина

- r ∈ V - начален връх
- adj = списък съседи
- eyes = {r} // тип стек - вкарваме отзад, вадим отзад
- visited = {}

- while #eyes > 0:
    - w ← eyes
    - eyes ← eyes - { w }

    - visited[w] = T

    - ∀w' ∈ adj[w]:
        - if not visited:
            - eyes ← eyes ∪ { w' }
